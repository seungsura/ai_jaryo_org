C → Java

C

#include <stdio.h>
#include <stdlib.h>

...

int main() {
    Point* p = (Point*)malloc(sizeof(Point));
    p->x = 10;
    p->y = 20;
    printf(
        "Point: %d, %d\n",
        p->x,
        p->y
    );
    free(p);
    return 0;
}
→

Java

Point p = new Point(10, 20);
System.out.println(
    "Point: " + p.x + ", " + p.y
);
p = null;
